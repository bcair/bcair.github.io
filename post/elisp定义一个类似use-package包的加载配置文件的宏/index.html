(let ((a "string A")
      (b "string B"))
  ;; m/test0 实际上等于 (symbol-value 'x), 不论传递的参数是什么
  (defmacro m/test0 (x)
    'x)
  (ignore-error 'void-variable (m/test0 b)) ;; => nil
  ;; 上一表达式返回为 nil, 说明捕获到 void-variable 这类错误

  (defmacro m/test1 (x)
    `'(,x . ,x))
  (macroexpand '(m/test1 a)) ;; => '(a . a)
  (m/test1 a)

  ;; m/test2 相比 m/test1 展开式会对 symbol 求值，然后组成 cons cell
  ;; m/test1 使用了 quote 宏来避免求值这一步
  (defmacro m/test2 (x)
    `(cons ,x ,x))
  (macroexpand '(m/test2 a))
  (m/test2 a) ;; => ("string A" . "string A")

  ;; 使用 quote 宏，则与 test1 表现一致
  (defmacro m/test2-quoted (x)
    `(cons ',x ',x))
  (macroexpand '(m/test2-quoted a))
  (m/test2-quoted vailete) ;; => (vailete . vailete)

  ;; 从 m/test1 和 m/test2-quoted 来看，一个反 quoted 符号 + quoted 符号
  ;;   `'(elt1 elt2 elt3)
  ;; 这种格式可以使到宏展开后的执行结果与宏展开式一致，借此我们可以用来看下面的例子：
  ;;   传递给宏的参数不再是 symbol, 而是 cons 类型
  (defmacro m/test3 (x)
    (print (symbolp x))                 ; nil
    (print (type-of x))                 ; cons
    (print x)                           ; (+ 400 200)
    `'(print (concat ,a (number-to-string ,x))))

  ;; 执行结果与返回值如下
  (m/test3 (+ 400 200))
  ;; (print (symbolp x))                 ; nil
  ;; (print (type-of x))                 ; cons
  ;; (print x)                           ; (+ 400 200)
  ;; => (print (concat "string A" (number-to-string (+ 400 200))))
  )

第一个表达式 (m/test a) 发送的消息中确定参数与 'a 等同，说明在宏展开时参数只考虑其打印形式即可；在函数运行时表达式中，为了再次验证是否与 'a 相等，需要将 ,x 用单引号标记为引用。

(defmacro fn/require-ver1 (&rest args)
  (declare (indent 1) (debug t))
  "Require config files. Possible keywords see follows:

:last (no supplement)        is it last config?
:path                        \"/path/to/dir/\"
:autoload                    (\"/path/to/file\" docstring interactive type)
:foo-map                     key-sequence"
  (let ((form '(progn)) pre-config lastp)
    (catch 'break
      (while t
        ;; do pcase filter in each while loop
        (pcase (pop args)
          ;; ends while structure
          ((and (pred null)
                (guard (= (length args) 0)))
           (throw 'break nil))

          ;; lastq (like the SEQ in tcp protocol), it would be checked in last clause
          (:last (setq lastp t))

          ((and :path
                (let newpath (pop args)))
           ;; newpath can be string or form which return a path string
           ;; the “,newpath” in next line will eval newpath and insert
           ;; result string
           (push `(add-to-list 'load-path ,newpath) form))

          ;; autoload 参数过多，将参数集中在关键字后面的列表中
          ;; 'func & :autoload
          ((and (pred consp)
                func-sym
                (let func func-sym)))
          ((and :autoload
                (let auto-load-args (pop args)))
           (push `(autoload ,func ,@auto-load-args) form))

          ;; handle the MAP written in keywords
          ((and (pred keywordp)
                map-keyword
                (let map-name (funcall
                               (lambda (x)
                                 (save-match-data
                                   (string-match "\\(?:\:\\)\\(.*map\\)$" x)
                                   (substring x (match-beginning 1) (match-end 1))))
                               map-keyword)))
           (if (string-prefix-p "global" map-name)
               '(define-key keymap key def)
             (with-eval-after-load 'after
               (define-key keymap key def))))

          ;; 1) Obtain new feature's symbol or file-name be loaded
          ((and (or (pred symbolp) (pred stringp))
                config
                (let config (if lastp pre-config config)) ; reset config when lastp is t
                )
           (if lastp
               (progn (push `(message "End at %s loading" ,config) form)
                      ;; also break current while loop
                      (throw 'break nil))
             (if (stringp config)
                 (push `(load ,config nil t) form)
               (push `(require ',config) form))
             ;; update pre-config for next while loop
             (setq pre-config config))))))
    (setq form (reverse form))
    ;; (nreverse form)
    form))

(defmacro m/handle (&rest args)
  ;; give handle-func args (in macro perspective) list
  (handle-func args nil))

;; env:
;;   target:  cur-feature, load-file-name, or autoload-function
;;   extra1:  after-feature autoload-file-name keymap-name keyseq

;; 所以 env 结构为
;;   '(&rest charged type value after autoload keymap keyseq)
;; charged 为填充状态，每个 target 参数没有过滤完毕时该变量为 nil, 一旦
;; charged 为 t 则生成一次宏展开代码

;; target match 逻辑
;; 1. 设置 charged 为 t
;; 2. 检查 type
;; 3. type 非空，生成代码
;; 4. type 空，跳过
;; 5. 生成代码后，清空 env
;; 6. 处理 EXPRVALs (pcase 逻辑)

;; autoload 注册函数情景，可能需要设置快捷键，即对应的 map, 若 map 默认
;; 不加载，添加 :after 关键字

(defun handle-func (x &rest env)
  (pcase (list (car x) (pop x) (car x))
    ;; EXPRVAL explain:
    ;;   '(PRED MAYBE-LET-PREV LET extra)
    ;;   notice: the argument of PRED & MAYBE-LET-PREV are equal
    ;;   PRED do checking, MAYBE-LET-PREV do bind or nothing, LET do binding

    (`(:path     ,_ ,path-str)  (progn (pop x)
                                       (handle-func )))
    (`(:autoload ,_ ,file-name) (progn (pop x)
                                       `(autoload func file-name nil t))) ; only command
    (`(:after    ,_ ,after)     (progn (pop x)
                                       `(with-eval-after-load ,after-feat))) ; only command
    (`(,(and (pred keywordp)
             (pred (lambda (y) (string-match "\\(?:\:\\)\\(.*map\\)$" (symbol-name y)))))
       ,(app (lambda (y) (substring y (match-beginning 1) (match-end 1))) map-name)
       ,keyseq)
     ;; map-name is a string, so need intern-soft
     (progn (pop x) (define-key (intern-soft map-name)))
     )

    (`(,(pred symbolp) ,feat ,_)  (handle-func (x `(feature   ,feat))))
    (`(,(pred stringp) ,file ,_)  (handle-func (x `(load-file ,file))))
    (`(,(and (pred consp)
             (pred (lambda (y) (eq (car y) 'quote))))))
    (`(nil nil nil) 'break))
  )

(progn
  (autoload 'func1 "foo" nil t nil)

1.2.6 关于 uninterned symbol

绑定键位时，有时因为配置未加载，需要的 map 可能没有 interned. 这是需要了解一些 uninterned symbol 以应对我们获取了 map 的字符串名(不是 symbol 类型)

因为： define-key 要求第一个参数为有效变量，且类型为 (unquoted) symbol

No obarray contains all symbols; in fact, some symbols are not in any obarray. They are called “uninterned symbols”. An uninterned symbol has the same four cells as other symbols; however, the only way to gain access to it is by finding it in some other object or as the value of a variable.

uninterned symbol 可以保存到变量中，为了更快键的创建 uninterned symbol, 可以执行

1.2.7 宏内判断参数为 quoted symbol

注意下面定义了 m/ts 后，只有参数打印形式为 quoted-symbol 才会返回真。比如下面的第三个 let 例子，因为 m/ts 为宏，参数 foo 不会进行求值，所以返回 nil.

(defmacro m/ts (x)
  (and (consp x)
       (funcall (lambda (y) (eq (car y) 'quote)) x)))

;; => t
(m/ts foo)
;; => nil
(let ((foo 'test))
  (m/ts foo))
;; => nil
